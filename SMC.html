// ═══════════════════════════════════════════ // 1. INDEXEDDB PERSISTENCE (200K TICKS CAPACITY) // ═══════════════════════════════════════════ const DB_NAME = "XAUUSD_SMC_DB"; const STORE_NAME = "TickHistory"; let db; async function initStorage() { return new Promise((resolve) => { const request = indexedDB.open(DB_NAME, 1); request.onupgradeneeded = (e) => { db = e.target.result; if (!db.objectStoreNames.contains(STORE_NAME)) { db.createObjectStore(STORE_NAME, { keyPath: "t" }); } }; request.onsuccess = (e) => { db = e.target.result; resolve(); }; }); } async function persistTick(price) { const tick = { t: Date.now(), p: price }; const tx = db.transaction(STORE_NAME, "readwrite"); const store = tx.objectStore(STORE_NAME); store.add(tick); // UI Update για το μέγεθος της βάσης const countRequest = store.count(); countRequest.onsuccess = () => { document.getElementById('dbCount').textContent = countRequest.result.toLocaleString(); // Εδώ μπορείς να προσθέσεις logic για σβήσιμο παλαιότερων αν θες αυστηρό limit }; } // ═══════════════════════════════════════════ // 2. ΔΙΟΡΘΩΜΕΝΗ SMC ΛΟΓΙΚΗ (HIGH/LOW OBs & GOLD ATR) // ═══════════════════════════════════════════ function detectOrderBlocks(candles) { const obs = []; if (!candles || candles.length < 5) return obs; for (let i = 2; i < candles.length - 1; i++) { const prev = candles[i-1]; // Το "Candidate" OB κερί const current = candles[i]; // Το κερί που κάνει το Breakout (BOS) // BULLISH OB: Το High του breakout κεριού κλείνει πάνω από το High του προηγούμενου πτωτικού if (current.c > prev.h && prev.c < prev.o) { const atr = calcATR(candles.slice(0, i)); const expansion = current.c - prev.h; if (expansion > atr * 0.5) { // Ισχυρή κίνηση (Displacement) obs.push({ type: 'BULL', top: prev.h, // Διόρθωση: Ολόκληρο το High/Low range bottom: prev.l, mid: (prev.h + prev.l) / 2, strength: Math.min(100, Math.round((expansion / atr) * 40)), t: prev.t }); } } // BEARISH OB: Το Low του breakout κεριού κλείνει κάτω από το Low του προηγούμενου ανοδικού if (current.c < prev.l && prev.c > prev.o) { const atr = calcATR(candles.slice(0, i)); const expansion = prev.l - current.c; if (expansion > atr * 0.5) { obs.push({ type: 'BEAR', top: prev.h, // Διόρθωση: Ολόκληρο το High/Low range bottom: prev.l, mid: (prev.h + prev.l) / 2, strength: Math.min(100, Math.round((expansion / atr) * 40)), t: prev.t }); } } } return obs.slice(-5); // Κρατάμε τα 5 πιο πρόσφατα } // ═══════════════════════════════════════════ // 3. GOLD VOLATILITY SCALING (ATR FIX) // ═══════════════════════════════════════════ function getGoldVolatilityScale(tf) { // Προσαρμογή του volMap για τις πραγματικές κινήσεις του XAUUSD const goldVol = { M1: 0.85, // ~85 pips avg move M5: 1.80, // ~1.8$ avg move M15: 3.50, H1: 8.00, H4: 15.00, D1: 28.00 }; return goldVol[tf] || 2.0; } // ═══════════════════════════════════════════ // 4. INTEGRATION ΜΕ ΤΟ SCRAPER ΣΟΥ // ═══════════════════════════════════════════ // Αυτή τη συνάρτηση την καλείς από το δικό σου DOM Scraper function handleScrapedPrice(price) { if (!price || isNaN(price)) return; // 1. Αποθήκευση στη μόνιμη βάση (IndexedDB) persistTick(price); // 2. Ενημέρωση του UI updatePriceDisplay(price); // 3. Εκτέλεση Ανάλυσης runSMCAnalysis(price); } function updatePriceDisplay(price) { const el = document.getElementById('livePrice'); const lastPrice = parseFloat(el.textContent) || 0; el.textContent = price.toFixed(2); el.style.color = price >= lastPrice ? "var(--green)" : "var(--red)"; // Tick Flash Effect el.classList.add('flash'); setTimeout(() => el.classList.remove('flash'), 200); }
